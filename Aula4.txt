/*Cria tabela de pessoas*/
CREATE TABLE IF NOT EXISTS  pessoa  (
   id_pessoa  INT NOT NULL AUTO_INCREMENT,
   nome  VARCHAR(100) ,
   nascimento  DATE ,
   sexo  ENUM('M', 'F') ,
   email  VARCHAR(200) NOT NULL,
  PRIMARY KEY ( id_pessoa )
);

/*Coloca email como unico*/
alter table pessoa add unique (email);
